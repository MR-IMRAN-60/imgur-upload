```bash
> npm i imgur-upload-apis
```


```js

const { Imgur } = require('imgur-upload-apis');

(async () => {
  try {
    const result = await Imgur('https://example.com/image.jpg');
    console.log(result);
  } catch (error) {
    console.error(error.message);
  }
})();
```
